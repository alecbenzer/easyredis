generate a ‘score’ for a string used for storing it in a sorted set

# File lib/easy_redis.rb, line 18
def self.string_score(str)
  str = str.downcase
  mult = 1.0
  scr = 0.0
  str.each_byte do |b|
    mult /= 27
    scr += (b-'a'.ord+1)*mult
  end
  scr
end
