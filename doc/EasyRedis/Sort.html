def initialize(field,order,klass)
  raise EasyRedis::FieldNotSortable, field  unless klass.sortable?(field) 
  raise EasyRedis::UnknownOrderOption, order  unless [:asc,:desc].member? order
  @field = field
  @order = order
  @klass = klass
end
